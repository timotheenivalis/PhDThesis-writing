\hlstd{FunctionGvalues}\hlkwb{<-}\hlkwa{function}\hlstd{(}\hlkwc{nbLoci}\hlstd{=}\hlnum{10}\hlstd{,}\hlkwc{nbAlleles}\hlstd{=}\hlnum{10}\hlstd{,}\hlkwc{dominance}\hlstd{=}\hlnum{0.5}\hlstd{,}\hlkwc{overdominance}\hlstd{=}\hlnum{0}\hlstd{,}\hlkwc{SDeffects}\hlstd{=}\hlnum{1}\hlstd{,}\hlkwc{SDalleles}\hlstd{=}\hlnum{1}\hlstd{)}
\hlstd{\{}
  \hlstd{gvalues}\hlkwb{<-}\hlkwd{array}\hlstd{(}\hlkwc{data}\hlstd{=}\hlnum{NA}\hlstd{,}\hlkwc{dim}\hlstd{=}\hlkwd{c}\hlstd{(nbAlleles,nbAlleles,nbLoci),}\hlkwc{dimnames}\hlstd{=}\hlkwd{list}\hlstd{(}\hlkwd{paste}\hlstd{(}\hlstr{"A"}\hlstd{,}\hlnum{1}\hlopt{:} \hlstd{nbAlleles,}\hlkwc{sep}\hlstd{=}\hlstr{""}\hlstd{),}\hlkwd{paste}\hlstd{(}\hlstr{"A"}\hlstd{,}\hlnum{1}\hlopt{:} \hlstd{nbAlleles,}\hlkwc{sep}\hlstd{=}\hlstr{""}\hlstd{),}\hlkwd{paste}\hlstd{(}\hlstr{"L"}\hlstd{,}\hlnum{1}\hlopt{:}\hlstd{nbLoci,}\hlkwc{sep}\hlstd{=}\hlstr{""}\hlstd{)))} \hlcom{# Initialising a matrix that will contain the genotypic effects on the/a trait}
  \hlkwa{for}\hlstd{(L} \hlkwa{in} \hlnum{1}\hlopt{:}\hlstd{nbLoci)}
  \hlstd{\{}
    \hlcom{# Setting the effects for the homozygotes [all loci]}
    \hlstd{effect}\hlkwb{<-}\hlkwd{abs}\hlstd{(}\hlkwd{rnorm}\hlstd{(}\hlkwc{n}\hlstd{=}\hlnum{1}\hlstd{,}\hlkwc{mean}\hlstd{=}\hlnum{0}\hlstd{,}\hlkwc{sd}\hlstd{=SDeffects))}\hlcom{# alter the locus importance in a realistic way (many small-effect loci, few major loci)}
    \hlkwd{diag}\hlstd{(gvalues[,,L])}\hlkwb{<-}\hlnum{2}\hlopt{*}\hlkwd{rnorm}\hlstd{(}\hlkwc{n}\hlstd{=}\hlkwd{dim}\hlstd{(gvalues)[}\hlnum{1}\hlstd{],}\hlkwc{mean}\hlstd{=}\hlnum{0}\hlstd{,}\hlkwc{sd}\hlstd{=effect}\hlopt{*}\hlstd{SDalleles)}
    \hlcom{# Setting the effects for the heterozygotes}
    \hlkwa{for}\hlstd{(A} \hlkwa{in} \hlnum{1}\hlopt{:}\hlstd{(nbAlleles}\hlopt{-}\hlnum{1}\hlstd{))}\hlcom{# loop for off-diagonal = heterozygotes (additive and dominance effects)}
    \hlstd{\{}
      \hlkwa{for} \hlstd{(D} \hlkwa{in} \hlstd{(A}\hlopt{+}\hlnum{1}\hlstd{)}\hlopt{:}\hlstd{nbAlleles)}
      \hlstd{\{}
        \hlstd{d}\hlkwb{<-}\hlstd{dominance}\hlopt{*}\hlkwd{runif}\hlstd{(}\hlkwc{n}\hlstd{=}\hlnum{1}\hlstd{,}\hlkwc{min}\hlstd{=}\hlopt{-}\hlnum{0.5}\hlopt{-}\hlstd{overdominance,}\hlkwc{max}\hlstd{=}\hlnum{0.5}\hlopt{+}\hlstd{overdominance)}
        \hlstd{gvalues[A,D,L]}\hlkwb{<-}\hlstd{(}\hlnum{0.5}\hlopt{-}\hlstd{d)}\hlopt{*}\hlstd{gvalues[A,A,L]}\hlopt{+}\hlstd{(}\hlnum{0.5}\hlopt{+}\hlstd{d)}\hlopt{*}\hlstd{gvalues[D,D,L]} \hlcom{# mean of additive effects + dominance, over diagonal}
        \hlstd{gvalues[D,A,L]}\hlkwb{<-}\hlstd{(}\hlnum{0.5}\hlopt{-}\hlstd{d)}\hlopt{*}\hlstd{gvalues[A,A,L]}\hlopt{+}\hlstd{(}\hlnum{0.5}\hlopt{+}\hlstd{d)}\hlopt{*}\hlstd{gvalues[D,D,L]} \hlcom{# the same below diagonal    }
      \hlstd{\}}
    \hlstd{\}}
  \hlstd{\}}
  \hlkwd{return}\hlstd{(gvalues)}
\hlstd{\}}
